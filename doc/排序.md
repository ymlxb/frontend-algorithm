## 冒泡排序
```js
function compare(a,b){
  if(a>b) return true
  else return false

}
function exchange(arr,a,b) { // 将数组中 ab 位置里的值进行交换
   let temp = arr[a]
   arr[a] = arr[b]
   arr[b] = temp
}
function sort(arr){
  for(let i =0;i<arr.length;i++){
    for(let j = 0;j<arr.length-1-i;j++){
      if(compare(arr[j],arr[j+1])){
        exchange(arr,j,j+1)
      }
    }
  }
}
```
