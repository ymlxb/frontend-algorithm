深度优先搜索就跟二叉树的前序遍历一样

技巧：
    1. 访问根节点
    2. 对根节点的children挨个进行深度优先搜索

### 广度优先遍历

从根出发，优先访问离根节点最近的节点
技巧：
    1. 新建一个队列，把根节点入队
    2. 把队头出队
    3. 把队头的children挨个入队
    4. 重复2，3直到队列为空

```js
const fun2 = (root) =>{
  const arr = [root]
  while(arr.length > 0){
    cosnt n = arr.shift()
    n.children.forEach(item => {
      arr.push(item)
    })
  }
}
fun2(tree)
```

### 前序遍历

```js
/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number[]}
 */
var preorderTraversal = function(root) {
    let arr = []
    const fun = node => {
        if(node){
            arr.push(node.val)
            fun(node.left)
            fun(node.right)
        }
    }
    fun(root)
    return arr
};
```

### 二叉树的最小深度

思路：利用广度优先搜索来逐层遍历树的节点，一旦遇到叶子结点(没有左右子节点的节点)，当前的深度就是最小深度。

```js
var minDepth = function(root) {
    if (!root) return 0; // 处理空树的情况
    
    const stack = [[root, 1]]; // 初始化队列，每个元素是 [节点, 深度]
    
    while (stack.length) {
        const [node, depth] = stack.shift(); // 取出队列头部元素
        
        if (!node.left && !node.right) { // 检查是否为叶子节点
            return depth; // 找到最小深度，立即返回
        }
        
        // 将子节点加入队列，深度加1
        if (node.left) stack.push([node.left, depth + 1]);
        if (node.right) stack.push([node.right, depth + 1]);
    }
    
};
```

### 二叉树的最大深度

思路：使用广度优先搜索（BFS）逐层遍历，记录层数。使用队列进行 BFS 遍历，每次处理一层的所有节点并将下一次的节点加入队列，每处理完一次，深度加1

```js
/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var maxDepth = function(root) {
    if(!root) return 0
    const stack = [root]
    let depth = 0
    while(stack.length){
        let length = stack.length
        while(length--){
            const node = stack.shift()
            if(node.left) stack.push(node.left)
            if(node.right) stack.push(node.right)
        }
        depth++
    }
    return depth
};
```

### 中序后序还原二叉树
```javascript
let end = ['f', 'g', 'c', 'd', 'e', 'b', 'a'];
let middle = ['f', 'c', 'g', 'a', 'd', 'b', 'e']

function Node(value){
  this.value = value
  this.left = null
  this.right = null
}

function f1(end, middle) {
  if(end == null || middle ==null || end.length == 0 || middle.length == 0 || end.length != middle.length) return null;
  let root = new Node(end[end.length-1])
  let index = middle.indexOf(root.value)
  let endLeft =end.slice(0,index)
  let endRight = end.slice(index,end.length-1)
  let middleLeft = middle.slice(0,index)
  let middleRight = middle.slice(index + 1,middle.length)
  root.left = f1(endLeft,middleLeft)
  root.right = f1(endRight,middleRight)
  return root
}

let root = f1(end,middle)
console.log(root.left);
console.log(root.right);
```
