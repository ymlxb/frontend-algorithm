function add () {
  let args = Array.prototype.slice.call(arguments);
  // 该函数会被返回，该函数的作用是继续收集参数
  function _adder() {
    args.push(...arguments)
    return _adder;
  }
  // 当调用 toString 方法的时候说明我不要再接收参数了
  // 执行计算操作
  _adder.toString = function () {
    return args.reduce((a, b) => a + b)
  }
  return _adder;
}

console.log(add(1)(2)(3).toString());
console.log(add(1,2,3)(4).toString());
console.log(add(1)(2)(3,4,5).toString());

function curry(fn, ...args) {
  // fn.length：函数期望的参数个数（形参数量）
  // args.length：已经收集到的参数个数
  // 如果收集的参数 >= 函数期望的参数，就执行函数
  // 否则返回一个新函数继续收集参数
  
  if (fn.length <= args.length) {
    return fn(...args);  // 注意这里是展开参数！
  } else {
    // 返回一个新函数，这个函数会收集更多参数
    return function(...nextArgs) {
      // 合并已收集的和新传入的参数
      return curry(fn, ...args, ...nextArgs);
    };
  }
}
